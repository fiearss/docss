    ka/ui/dropdown/Dropdown.js

    /**
 * Класс Dropdown представляет собой настраиваемый выпадающий список.
 * Он состоит из кнопки-триггера и контента, который скрывается/появляется по клику на триггер.
 * Контент состоит из чекбоксов, которые могут быть добавлены динамически.
 */
export default class Dropdown {
    /**
     * Конструктор Dropdown инициализирует новый экземпляр выпадающего списка.
     * @param {string|Element} selectorOrElement Строка с селектором или элемент, в который нужно добавить выпадающий список.
     * @param {string} caption Подпись кнопки-триггера.
     */
    constructor(selectorOrElement, caption) {
        // Создание DOM-элемента выпадающего списка
        if (typeof selectorOrElement === 'string') {
            this.container = document.querySelector(selectorOrElement);
        } else if (selectorOrElement instanceof Element) {
            this.container = selectorOrElement;
        } else {
            throw new Error('Селектор или элемент ожидается');
        }
        this.container.classList.add('dropdown');

        // Создание кнопки-триггера
        this.button = document.createElement('button');
        this.button.classList.add('dropdown__button');
        this.button.textContent = caption;
        this.button.addEventListener('click', (event) => {
            event.stopPropagation();
            this.toggleDropdown();
        });

        // Создание контента
        this.content = document.createElement('div');
        this.content.classList.add('dropdown__content');

        // Добавление элементов в контейнер
        this.container.appendChild(this.button);
        this.container.appendChild(this.content);

        // Добавление слушателя клика на документ для закрытия выпадающего списка при клике вне
        document.addEventListener('click', (event) => {
            this.handleClickOutside(event);
        });
    }

    /**
     * Метод toggleDropdown переключает отображение контента выпадающего списка.
     */
    toggleDropdown() {
        this.container.classList.toggle('dropdown--open');
    }

    /**
     * Метод closeDropdown скрывает выпадающий список.
     */
    closeDropdown() {
        this.container.classList.remove('dropdown--open');
    }

    /**
     * Метод addFromDict добавляет все элементы из объекта в контент.
     * @param {Object} dict Объект, содержащий значения и их HTML-представление.
     */
    addFromDict(dict) {
        for (const [value, html] of Object.entries(dict)) {
            this.addItem(value, html);
        }
    }

    /**
     * Метод setFromDict очищает контент и добавляет все элементы из объекта.
     * @param {Object} dict Объект, содержащий значения и их HTML-представление.
     */
    setFromDict(dict) {
        this.content.innerHTML = '';
        this.addFromDict(dict);
    }

    /**
     * Метод addItem добавляет новый элемент в контент.
     * @param {string} value Значение чекбокса.
     * @param {string} html HTML-представление чекбокса.
     */
    addItem(value, html) {
        const label = document.createElement('label');
        label.classList.add('dropdown__label');
        label.innerHTML = `<input type="checkbox" value="${value}"> ${html}`;
        label.querySelector('input').addEventListener('change', () => this.handleChange());
        this.content.appendChild(label);
    }

    /**
     * Метод handleChange вызывается при изменении состояния чекбокса и объявляет событие dropdownChange.
     */
    handleChange() {
        const event = new CustomEvent('dropdownChange', { detail: this.getValues() });
        this.container.dispatchEvent(event);
    }

    /**
     * Метод getValues возвращает массив значений выбранных чекбоксов.
     * @returns {Array} Массив значений выбранных чекбоксов.
     */
    getValues() {
        return Array.from(this.content.querySelectorAll('input:checked')).map(input => input.value);
    }

    /**
     * Метод onDropdownChange привязывает обработчик события dropdownChange к контейнеру.
     * @param {function} callback Функция-обработчик события dropdownChange.
     */
    onDropdownChange(callback) {
        this.container.addEventListener('dropdownChange', callback);
    }

    /**
     * Метод handleClickOutside обрабатывает событие клика на документ.
     * Если клик был вне контейнера, выпадающий список закрывается.
     * @param {Event} event Объект события клика.
     */
    handleClickOutside(event) {
        if (!this.container.contains(event.target)) {
            this.closeDropdown();
        }
    }

    /**
     * Метод selectAll выбирает все чекбоксы в контенте.
     */
    selectAll() {
        this.content.querySelectorAll('input:not(:checked)').forEach(input => input.checked = true);
        this.handleChange();
    }

    /**
     * Метод unselectAll снимает выбор со всех чекбоксов в контенте.
     */
    unselectAll() {
        this.content.querySelectorAll('input:checked').forEach(input => input.checked = false);
        this.handleChange();
    }
}
