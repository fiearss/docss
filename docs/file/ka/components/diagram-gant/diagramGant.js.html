    import dateUtils from "../../utils/dateUtils.js";
// TODO: Переписать все методы через DiagramGantBase и удалить этот файл


/**
 * Класс DiagramGant представляет собой компонент диаграммы Ганта.
 * 
 * Диаграмма Ганта - это графическое представление процессов или задач с
 * обозначенными временными интервалами. Компонент обеспечивает создание
 * диаграммы Ганта и ее настройку с помощью данных.
 */
export class DiagramGant {

    /**
     * Конструктор класса DiagramGant.
     *
     * @param {string} selector - Селектор для поиска элемента в DOM,
     * в котором будет размещаться диаграмма Ганта.
     * @param {string} diagramName - Название диаграммы.
     */
    constructor(selector, diagramName = '') {
        // Поиск элемента в DOM по указанному селектору.
        this.element = document.querySelector(selector);
        // Начальная и конечная даты периода.
        this.start = null;
        this.end = null;
        // Общая длительность периода.
        this.global_interval = null;
        // Флаг, указывающий на изменение данных диаграммы.
        this.isSetData = false;
        // Название диаграммы.
        this.diagramName = diagramName;
    }

    /**
     * Устанавливает начальную и конечную даты периода.
     *
     * @param {Date} start - Начальная дата периода.
     * @param {Date} end - Конечная дата периода.
     */
    setPeriod(start, end) {
        this.start = start;
        this.end = end;
        this.global_interval = this.end - this.start;
    }

    /**
     * Очищает внутреннее содержимое элемента и устанавливает флаг isChange в false.
     *
     * Этот метод очищает внутреннее содержимое элемента, на который ссылается this.element,
     * и устанавливает флаг isChange в false. Флаг используется для отслеживания,
     * были ли данные диаграммы изменены.
     *
     * @return {void} Этот метод не возвращает значения.
     */
    clear() {
        // Очищает внутреннее содержимое элемента, на который ссылается this.element.
        this.element.innerHTML = '';
        // Устанавливает флаг isChange в false.
        this.isSetData = false;
    }

    /**
     * Устанавливает данные для компонента.
     *
     * @param {Date} start - Начальная дата периода.
     * @param {Date} end - Конечная дата периода.
     * @param {Array} data - Данные, которые нужно установить.
     * @return {void}
     */
    setData(start, end, data) {
        this.clear();
        this.setPeriod(start, end);
        this.isSetData = true;
        this.setName(this.diagramName);
        this.addDateBlock(start, end);
    }

    /**
     * Добавляет блок с датой начала и датой окончания в диаграмму Ганта.
     *
     * @param {Date} start - Дата начала периода.
     * @param {Date} end - Дата окончания периода.
     */
    addDateBlock(start, end) {
        // Формируем текст подписи
        const formattedStart = dateUtils.formatDisplayDateTime(start); // Форматируем дату начала
        const formattedEnd = dateUtils.formatDisplayDateTime(end); // Форматируем дату конца
        const interval = Math.round((end - start) / 1000);
        const captionText = `За период c ${formattedStart} по ${formattedEnd}.&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;Продолжительность периода анализа: ${dateUtils.secondsToHMS(interval)} сек.`; // Объединяем все данные в одну строку
        const div = document.createElement('div');
        div.classList.add('diagram-gant__date');
        div.innerHTML = captionText;
        this.element.appendChild(div);
    }

    /**
     * Добавляет блок с сообщением "Нет данных" в диаграмму Ганта.
     *
     * @param {string} name - Название блока. По умолчанию "Нет данных".
     */
    setNoneData(name = 'Нет данных') {
        const div = document.createElement('div');
        div.classList.add('diagram-gant__none-data');
        div.innerHTML = name;
        this.element.appendChild(div);
    }

    /**
     * Создает дочерний div для диаграммы с ее названием.
     *
     * @param {string} divText - Название диаграммы.
     */
    setName(divText) {
        const diagramName = document.createElement('div');
        diagramName.classList.add('diagram-gant__name');
        diagramName.innerHTML = divText;
        this.element.appendChild(diagramName);
    }

    /**
     * Добавляет строку в диаграмму Ганти.
     * Добавляет строку в диаграмму Ганта.
     *
     * @param {HTMLElement} groupBlock - Элемент HTML, в который добавляется строка.
     * @param {string} name - Название строки.
     * @param {Date} start - Дата начала строки.
     * @param {Date} end - Дата конца строки.
     * @param {string} caption - Подпись строки.
     * @param {boolean} selected - Является ли строка выбранной или нет.
     * @param {string|null} bg - Цвет фона строки.
     * @param {boolean} transparent - Прозрачность строки.
     * @param {boolean} is_bold - Жирный шрифт.
     * @param {string} empty_name - Название строки, если она находится вне зоны КА.
     * @param {boolean} is_line_sep - Разделительная линия.
     * @param {string} style - Стили строки.
     */
    addRowGant(groupBlock, name, start, end, caption, selected = false, bg = null, transparent = false, is_bold = false, empty_name = 'Вне зоны КА', is_line_sep = false, style = '') {
        // Создание новой строки
        const line = this.addRow(groupBlock, name, is_bold, is_line_sep, style);


        // Добавление неактивного участка диаграммы, если он находится за пределами зоны КА
        if (this.start.getTime() !== start.getTime()) {
            this.addLineNull(line, this.start, start, empty_name);
        }

        // Добавление активного участка диаграммы
        this.addLineData(line, start, end, caption, selected, bg, transparent);

        // Добавление неактивного участка диаграммы, если он находится за пределами зоны КА
        if (this.end.getTime() !== end.getTime()) {
            this.addLineNull(line, end, this.end, empty_name);
        }

        // Добавление подписи к активному участку диаграммы
        this.addCaptionData(line, start, end, caption);
    }


    /**
     * Добавляет групповой блок на диаграмму.
     *
     * @param {string} name - Имя группы блока.
     * @return {HTMLElement} Созданный элемент группы блока.
     */
    addGroupBlock(name) {
        // Создаем новый блок для группы
        const groupBlock = document.createElement('div');
        groupBlock.classList.add('diogram-gant__abonent');

        const nameDiv = document.createElement('div');
        nameDiv.classList.add('diogram-gant__row-name');
        nameDiv.innerHTML = name;
        groupBlock.append(nameDiv);
        this.element.append(groupBlock);
        return groupBlock;
    }

    /**
     * Добавляет строку в указанный групповой блок с заданным именем. `row-sputnik`
     *
     * @param {HTMLElement} groupBlock - Групповой блок, в который будет добавлена строка.
     * @param {string} name - Имя строки.
     * @return {HTMLElement} Созданный элемент строки `diogram-gant__row-interval`.
     */
    addRow(groupBlock, name, bold = false, is_line_sep = false, name_style = '') {
        // Создаем блок для названия строки
        const nameBlock = document.createElement('div');
        nameBlock.classList.add('diagram-gant__row-sputnik');
        nameBlock.innerHTML = name;

        // Создаем блок строки
        const line = document.createElement('div');
        line.classList.add('diagram-gant__row-interval', 'row-interval', 'row-interval--procent');

        // Устанавливаем стиль для названия строки
        if (bold) {
            nameBlock.style.fontWeight = 'bold';
        } else {
            nameBlock.style.fontSize = '14px';
        }

        // Устанавливаем стиль для разделительной линии
        if (is_line_sep) {
            line.classList.add('diagram-gant__end-rows-block');
        }

        // Применяем пользовательский стиль
        if (name_style) {
            nameBlock.style.cssText += name_style;
        }

        // Добавляем блоки в групповой блок
        groupBlock.append(nameBlock, line);
        return line
    }

    /**
     * Добавляет строку с данными в указанный элемент строки.
     *
     * @param {HTMLElement} line - Элемент строки, в который будет добавлена строка с данными.
     * @param {number} start - Начальное время данных.
     * @param {number} end - Конечное время данных.
     * @param {string} [name=''] - Имя данных. По умолчанию - пустая строка.
     * @param {boolean} [selected=false] - Являются ли данные выбранными или нет. По умолчанию - false.
     */
    addLineData(line, start, end, name = '', selected = false, bg = null, transparent = false) {
        // Вычисляем разницу между конечным и начальным временем
        const interval = end - start;

        // Вычисляем долю временного интервала в отношении общей длительности данных
        const ration = interval / this.global_interval;

        // Создаем блок для данных
        const block = document.createElement('div');
        block.classList.add('row-interval__block', 'row-interval__block--in');

        // Устанавливаем стиль для выбранных данных
        if (selected) {
            block.classList.add('row-interval__block--selected');
        }

        // Устанавливаем стиль фона для данных
        if (bg) {
            block.style.background = bg;
        }

        // Устанавливаем стиль для прозрачных данных
        if (transparent) {
            block.classList.add('row-interval__block--transparent');
        }

        // Устанавливаем ширину блока в процентах
        block.style.width = `${ration * 100}%`;

        // Добавляем текст в блок
        this.addTextSpan(block, name + " сек.");

        // Добавляем блок в строку
        line.append(block);
    }

    /**
     * Добавляет текст в блок данных.
     *
     * @param {HTMLElement} block - Блок данных.
     * @param {string} text - Текст для добавления.
     */
    addTextSpan(block, text) {
        const span = document.createElement('span');
        span.classList.add('row-interval__value-text');
        span.innerHTML = text;
        span.title = text;
        block.append(span);
    }

    /**
     * Добавляет подпись с информацией о временных интервалах в указанный элемент строки.
     *
     * @param {HTMLElement} line - Элемент строки, в который будет добавлена подпись.
     * @param {Date} start - Дата начала временного интервала.
     * @param {Date} end - Дата конца временного интервала.
     * @param {number} interval - Длительность временного интервала в секундах.
     */
    addCaptionData(line, start, end, interval) {
        // Создаем элемент подписи
        const caption = document.createElement('div');
        caption.classList.add('row-interval__caption');

        // Формируем текст подписи
        const formattedStart = dateUtils.formatDisplayDateTime(start); // Форматируем дату начала
        const formattedEnd = dateUtils.formatDisplayDateTime(end); // Форматируем дату конца
        const captionText = `c ${formattedStart} по ${formattedEnd}. &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;Время доступности: ${interval} сек.`; // Объединяем все данные в одну строку

        // Устанавливаем текст подписи
        caption.innerHTML = captionText;

        // Добавляем подпись в строку
        line.append(caption);
    }

    /**
     * Добавляет элемент с информацией о промежутке времени, выходящего за пределы данных.
     *
     * @param {HTMLElement} line - Элемент строки, в который будет добавлен блок.
     * @param {Date} start - Дата начала временного интервала.
     * @param {Date} end - Дата конца временного интервала.
     * @param {string} name - Название блока.
     */
    addLineNull(line, start, end, name = 'Вне зоны КА') {
        // Количество секунд в промежутке времени
        const interval = end - start;
        // Доля промежутка времени в общей длительности данных
        const ration = interval / this.global_interval;
        // Создаем блок
        const block = document.createElement('div');
        // Добавляем классы для стилизации
        block.classList.add('row-interval__block', 'row-interval__block--out');
        // Устанавливаем ширину блока в процентах
        block.style.width = `${ration * 100}%`;
        // Вставляем текст в блок
        this.addTextSpan(block, name);
        // Добавляем блок в строку
        line.append(block);
    }
}

/**
 * Класс CallersOverlayGant наследуется от DiagramGant и отвечает за управление и отображение данных о звонящих и их спутниках в формате диаграммы Ганта.
 */
export class CallersOverlayGant extends DiagramGant {

    /**
     * Устанавливает данные для диаграммы Ганта.
     *
     * @param {Date} start - Начальная дата отображаемого периода.
     * @param {Date} end - Конечная дата отображаемого периода.
     * @param {Object} data - Объект, содержащий данные о звонящих и их спутниках.
     */
    setData(start, end, data) {
        // Вызываем метод родительского класса для установки данных
        super.setData(start, end, data);

        // Проверяем, пусты ли данные
        if (data.length === 0) {
            this.setNoneData();
            return;
        }

        const { callers_sats_data, first_acceptable_session, general_availability_of_sessions, sessions, united_satellite_data, callers_combined_availability } = data;

        // Проверяем, пусты ли данные о звонящих и спутниках
        if (callers_sats_data.length === 0 && united_satellite_data.satellites_communication_data.length === 0 && callers_combined_availability.length === 0) {
            this.setNoneData();
            return;
        }

        const united_satellite_dataDICT = {};

        // Создаем словарь для данных объединенных спутников
        for (const sat of united_satellite_data.satellites_communication_data) {
            const { name, BSSs_and_callers_general_availability, BSSs_general_availability, BSSs_satellites_general_availability } = sat;
            united_satellite_dataDICT[name] = {
                BSSs_and_callers_general_availability,
                BSSs_general_availability,
                BSSs_satellites_general_availability,
            };
        }

        const callers_combined_availabilityDICT = {};

        // Создаем словарь для данных объединенной доступности звонящих
        for (const sat of callers_combined_availability) {
            const { name, satellites_data, general_availability } = sat;
            callers_combined_availabilityDICT[name] = {
                satellites_data,
                general_availability
            };
        }

        // Проходимся по данным звонящих и добавляем строки для каждого звонящего и их спутников
        for (const caller of callers_sats_data) {
            const { name, satellites_data, general_availability } = caller;

            // Добавляем групповой блок для звонящего
            const groupBlock = this.addGroupBlock(name);

            // Получаем данные о всех спутниках и общей доступности для текущего звонящего
            const { satellites_data: all_sat_data, general_availability: all_gen_data } = callers_combined_availabilityDICT[name];

            // Проходимся по данным всех спутников и добавляем строки для каждого спутника
            for (const satGen of all_sat_data) {
                const { satellite_name, communication_interval } = satGen;
                // Добавляем строки для интервалов связи спутника
                this.general_row(
                    groupBlock,
                    communication_interval,
                    `${satellite_name} в зоне связи`,
                    'rgb(204, 240, 243)',
                    false
                );
            }

            // Добавляем общую доступность для объединенных данных звонящего
            this.general_row(groupBlock, all_gen_data, `${name} доступность КА`, '#a4e7c1', true, 'Вне зоны связи', false);
        }
    }

    /**
     * Добавляет строку с общими данными о доступности.
     *
     * @param {Object} groupBlock - Групповой блок для добавления строки.
     * @param {Array} general_availability - Массив данных о доступности.
     * @param {string} name - Имя строки.
     * @param {string|null} bg - Цвет фона строки.
     * @param {boolean} bold - Флаг, указывающий, должна ли строка быть жирной.
     * @param {string} empty_name - Имя для пустых интервалов.
     * @param {boolean} is_line_sep - Флаг, указывающий, должна ли строка быть разделенной.
     */
    general_row(groupBlock, general_availability, name, bg = null, bold = true, empty_name = 'Вне зоны связи', is_line_sep = true) {
        // Изначально выбранная дата равна начальной дате периода
        let selectedDate = this.start;

        // Добавляем строку без названия
        const line = this.addRow(groupBlock, name, bold, is_line_sep);

        // Итерируемся по каждой сессии в данных
        for (const session of general_availability) {
            // Создаем объекты даты начала и конца сессии
            const sessionStart = new Date(session.start_datetime_iso);
            const sessionEnd = new Date(session.end_datetime_iso);

            // Если дата начала сессии не равна начальной дате периода,
            // добавляем линию "Вне зоны КА"
            if (sessionStart.getTime() != selectedDate.getTime()) {
                this.addLineNull(line, selectedDate, sessionStart, empty_name);
            }

            // Добавляем строку данных сессии
            this.addLineData(line, sessionStart, sessionEnd, session.duration_in_sec, true, bg);

            // Обновляем выбранную дату до даты конца текущей сессии
            selectedDate = sessionEnd;
        }

        // Если выбранная дата не равна конечной дате периода,
        // добавляем линию "Вне зоны КА"
        if (selectedDate.getTime() !== this.end.getTime()) {
            this.addLineNull(line, selectedDate, this.end, empty_name);
        }
    }

/**
 * Класс SessionsDiagramGant наследуется от DiagramGant и отвечает за управление и отображение данных о сеансах связи в формате диаграммы Ганта.
 */
export class SessionsDiagramGant extends DiagramGant {
    /**
     * Устанавливает данные для компонента.
     *
     * @param {Date} start - Начальная дата периода.
     * @param {Date} end - Конечная дата периода.
     * @param {Array} data - Массив объектов. Каждый объект содержит следующие свойства:
     *   - `{string} first_caller_name` - Имя первого вызывающего.
     *   - `{string} second_caller_name` - Имя второго вызывающего.
     *   - `{string} first_sat_name` - Имя первого спутника.
     *   - `{string} second_sat_name` - Имя второго спутника.
     *   - `{string} start_datetime_iso` - Дата начала сессии в формате ISO.
     *   - `{string} end_datetime_iso` - Дата конца сессии в формате ISO.
     *   - `{number} duration_in_sec` - Длительность сессии в секундах.
     * @param {Object} [selectedData={}] - Объект с выбранными данными. Содержит следующие свойства:
     *   - `{string} first_caller_name` - Имя первого вызывающего.
     *   - `{string} second_caller_name` - Имя второго вызывающего.
     *   - `{string} first_sat_name` - Имя первого спутника.
     *   - `{string} second_sat_name_` - Имя второго спутника.
     */
    setData(start, end, data, selectedData = null) {
        // Вызываем родительский метод для установки данных.
        super.setData(start, end, data);

        // Если данные пусты, устанавливаем сообщение об отсутствии сеансов связи.
        if (data.length == 0) {
            this.setNoneData('Нет сеансов связи.');
            return;
        }

        let index = 1;

        // Итерируемся по каждой сессии в данных.
        for (const session of data) {
            let selected = false;

            // Проверяем, выбрана ли текущая сессия.
            if (selectedData) {
                selected = session.first_caller_name === selectedData.first_caller_name &&
                    session.second_caller_name === selectedData.second_caller_name &&
                    session.first_sat_name === selectedData.first_sat_name &&
                    session.second_sat_name === selectedData.second_sat_name;
            }

            // Создаем групповой блок с именами вызывающих.
            const groupBlock = this.addGroupBlock(`Сеанс ${index}`);
            index++;

            // Добавляем строку с деталями сессии.
            this.addRowGant(
                groupBlock,
                `${session.first_caller_name} - ${session.first_sat_name} <br> ${session.second_caller_name} - ${session.second_sat_name}`,
                new Date(session.start_datetime_iso),
                new Date(session.end_datetime_iso),
                session.duration_in_sec,
                selected, // Указываем, выбрана ли строка.
                null,
                false,
                false,
                'Вне зоны связи'
            );
        }

        // Отмечаем, что данные изменились.
        this.isChange = true;
    }
}

/**
 * Класс GeneralDiagramGant наследуется от DiagramGant и отвечает за управление и отображение данных о сеансах связи в общем формате диаграммы Ганта.
 */
export class GeneralDiagramGant extends DiagramGant {
    /**
     * Устанавливает данные для компонента.
     *
     * @param {Date} start - Начальная дата периода.
     * @param {Date} end - Конечная дата периода.
     * @param {Array} data - Массив объектов. Каждый объект содержит следующие свойства:
     *   - `{string} start_datetime_iso` - Дата начала сессии в формате ISO.
     *   - `{string} end_datetime_iso` - Дата конца сессии в формате ISO.
     *   - `{number} duration_in_sec` - Длительность сессии в секундах.
     */
    setData(start, end, data, callers_data) {
        // Вызываем родительский метод для установки данных.
        super.setData(start, end, data);

        // Проверяем, есть ли данные, если нет - устанавливаем сообщение об отсутствии сеансов связи.
        if (data.length == 0) {
            this.setNoneData('Нет сеансов связи.');
            return;
        }

        // Изначально выбранная дата равна начальной дате периода.
        let selectedDate = this.start;

        // Проходимся по всем абонентам и возвращаем text "A1+A2+A3..."
        const abonents_names = callers_data
            .map(item => item.name) // Извлекаем только значения свойства "name"
            .join('+'); // Объединяем их с разделителем "+"

        // Создаем групповой блок без названия.
        const groupBlock = this.addGroupBlock(abonents_names);

        // суммарная длительность всех сеансов.
        let totalDuration = 0;

        // Итерируемся по каждой сессии в данных.
        for (const session of data) {
            // Создаем объекты даты начала и конца сессии.
            const sessionStart = new Date(session.start_datetime_iso);
            const sessionEnd = new Date(session.end_datetime_iso);

            // Увеличиваем сумму длительности сессии
            totalDuration += session.duration_in_sec;

            // Обновляем выбранную дату до даты конца текущей сессии.
            selectedDate = sessionEnd;
        }


        // Символ пробела
        const space = '&nbsp;'

        // умножаем число пробелов
        const spaces = space.repeat(46);

        // Делаем caption
        const totalDurationString = spaces + 'Продолжительность периода связи ' + dateUtils.secondsToHMS(totalDuration);

        // Добавляем строку с caption
        const line = this.addRow(groupBlock, totalDurationString);

        // Обнуляем выбранную дату
        selectedDate = this.start;

        // Итерируемся по каждой сессии в данных.
        for (const session of data) {
            // Создаем объекты даты начала и конца сессии.
            const sessionStart = new Date(session.start_datetime_iso);
            const sessionEnd = new Date(session.end_datetime_iso);

            // Если дата начала сессии не равна начальной дате периода,
            // добавляем линию "Вне зоны связи".
            if (sessionStart.getTime() != selectedDate.getTime()) {
                this.addLineNull(line, selectedDate, sessionStart, 'Вне зоны связи');
            }

            // Добавляем строку данных сессии.
            this.addLineData(line, sessionStart, sessionEnd, session.duration_in_sec);

            // Обновляем выбранную дату до даты конца текущей сессии.
            selectedDate = sessionEnd;

            // Увеличиваем сумму длительности сессии
            totalDuration += session.duration_in_sec;
        }

        // Если выбранная дата не равна конечной дате периода,
        // добавляем линию "Вне зоны связи".
        if (selectedDate.getTime() !== this.end.getTime()) {
            this.addLineNull(line, selectedDate, this.end, 'Вне зоны связи');
        }
    }
