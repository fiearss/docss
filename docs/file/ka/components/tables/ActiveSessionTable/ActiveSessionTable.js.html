    ka/components/tables/ActiveSessionTable/ActiveSessionTable.js

    import dateUtils from "../../../utils/dateUtils.js";
import { ModalInfo } from "./ModalInfo.js";

import Table from "./Table.js";


export default class ActiveSessionTable {

    constructor(selector) {
        this.element = document.querySelector(selector);
        this.container = document.createElement('div');
        this.modal = new ModalInfo();
        this.element.appendChild(this.modal.getElement());
        this.element.appendChild(this.container);
    }

    clear() {
        this.container.innerHTML = '';
    }

    setData(dataSessions) {
        this.clear();
        dataSessions.forEach(session => {
            this.addSession(session);
        })
    }

    addSession(session) {
        const sessionData = session.session_data;
        const sessionBlock = this.createSessionBlock();
        sessionBlock.appendChild(this.createHeader(session.ID, sessionData));
        sessionBlock.appendChild(this.createMainTable(sessionData));
        const detailsTable = sessionBlock.appendChild(this.createDetailsTable(sessionData));
        sessionBlock.appendChild(this.createToggleButton(detailsTable));
        // sessionBlock.appendChild(this.createModalButton(detailsTable));

        this.container.appendChild(sessionBlock);
    }

    createSessionBlock() {
        const sessionBlock = document.createElement('div');
        sessionBlock.style.border = '1px solid black';
        sessionBlock.style.borderRadius = '10px';
        sessionBlock.style.marginBottom = '10px';
        sessionBlock.style.padding = '10px';

        return sessionBlock;
    }

    createHeader(ID, sessionData) {
        const div = document.createElement('div');
        div.style.textAlign = 'center';
        div.style.border = '1px solid black';
        div.style.borderRadius = '10px';
        div.style.backgroundColor = '#cccccc';
        div.style.marginBottom = '10px';
        div.style.width = '100%';
        div.innerHTML = `Сеанс связи: ${dateUtils.formatDisplayDateTime(new Date(sessionData['Data_Vyz']))}_${ID}`
        return div
    }

    createMainTable(data) {
        const table = new Table();
        // TODO: ПРОВЕРКА НА УБИРАЕНИЕ
        if (data['ID_Abonent_T1'] == 0 || data['ID_Abonent_T2'] == 0) {
            data['ID_Abonent_T1'] = 1;
            data['ID_Abonent_T2'] = 2;
        }
        const rows = [
            { name: 'Вызывающий Аб.:', value: `Аб. ${data['ID_Abonent_T1']}` },
            { name: 'Вызываемый Аб.:', value: `Аб. ${data['ID_Abonent_T2']}` },
            { name: 'Дата вызова:', value: dateUtils.formatDisplayDateTime(new Date(data['Data_Vyz'])) },
        ];
        // Стили таблицы
        table.tableStyles = {
            width: '100%',
            fontSize: '14px',
            borderCollapse: 'collapse',
            marginBottom: '10px',
            wordBreak: 'break-word',
            whiteSpace: 'normal',
            borderCollapse: 'separate', /* Позволяет использовать border-spacing */
            borderSpacing: '5px', /* Расстояние между ячейками таблицы */
        }
        // Стили ячейки с названием
        table.tdNameStyles = {
            width: '160px',
            maxWidth: '160px',
            minWidth: '160px',
        }
        // Стили ячейки со значением
        table.tdValueStyles = {
        }
        table.setData(rows);

        return table.getElement();
    }

    createDetailsTable(data) {
        const table = new Table();
        const rows = [
            { name: 'Запрос на сеанс связи:', value: data['ID_Zapros_Seans_Tek'] },


            { name: 'Номер телефона вызывающего:', value: data['Tlf1'] },
            { name: 'КА вызывающего:', value: data['ID_KA1'] },
            { name: 'РСС вызывающего:', value: data['ID_RSS1'] },

            { name: 'Номер канала вызывающего:', value: data['Canal1'] },
            { name: 'Номер тайм-слота вызывающего:', value: data['Time_Slot1'] },

            { name: 'Номер телефона вызываемого:', value: data['Tlf2'] },
            { name: 'КА вызываемого:', value: data['ID_KA2'] },
            { name: 'РСС вызываемого:', value: data['ID_RSS2'] },

            { name: 'Номер канала вызываемого:', value: data['Canal2'] },
            { name: 'Номер тайм-слота вызываемого:', value: data['Time_Slot2'] },

            { name: 'Дата вызова:', value: dateUtils.formatDisplayDateTime(new Date(data['Data_Vyz'])) },
            { name: 'Дата решения:', value: dateUtils.formatDisplayDateTime(new Date(data['Data_Otv'])) },
            { name: 'Дата начала сеанса:', value: dateUtils.formatDisplayDateTime(new Date(data['Data_Beg'])) },
            { name: 'Дата начала передачи:', value: dateUtils.formatDisplayDateTime(new Date(data['Data_Beg_Razg'])) },

            // { name: 'На удаление:', value: '' },
            { name: 'Дата завершения сеанса:', value: dateUtils.formatDisplayDateTime(new Date(data['Data_End'])) },
            { name: 'Длительность сеанса:', value: data['Time_Seans'] },
            { name: 'Длительность передачи:', value: data['Time_Razg'] },
            // Добавьте другие поля, если необходимо
        ];
        // Стили таблицы
        table.tableStyles = {
            width: '100%',
            borderCollapse: 'collapse',
            marginBottom: '10px',
            wordBreak: 'break-word',
            whiteSpace: 'normal',
            borderCollapse: 'separate', /* Позволяет использовать border-spacing */
            borderSpacing: '5px', /* Расстояние между ячейками таблицы */
            fontSize: '14px',
            display: 'none',
        }

        // Стили ячейки с названием
        table.tdNameStyles = {
            width: '160px',
            maxWidth: '160px',
            minWidth: '160px',
        }
        // Стили ячейки со значением
        table.tdValueStyles = {
            // border: '1px solid black',
        }
        table.setData(rows);

        return table.getElement();
    }

    createToggleButton(detailsTable) {
        const container = document.createElement('div');
        container.style.display = 'flex';
        container.style.justifyContent = 'flex-end'; // Выравнивание по правому краю
        container.style.marginBottom = '5px';
        // Кнопка "Показать подробно/Скрыть"
        const toggleButton = document.createElement('button');
        toggleButton.style.marginLeft = 'auto';
        toggleButton.style.cursor = 'pointer';
        toggleButton.style.fontSize = '12px';
        toggleButton.style.fontWeight = 'bold';
        toggleButton.style.width = '160px';
        toggleButton.style.backgroundColor = '#ada5a5';
        toggleButton.style.boxShadow = 'none';
        toggleButton.style.border = 'none';
        toggleButton.style.borderRadius = '5px';
        toggleButton.style.height = '20px';
        toggleButton.textContent = 'Детализация';
        toggleButton.addEventListener('click', () => {
            const isHidden = detailsTable.style.display === 'none';
            detailsTable.style.display = isHidden ? 'table' : 'none';
            toggleButton.textContent = isHidden ? 'Скрыть' : 'Детализация';
        });

        toggleButton.addEventListener('mouseover', () => {
            toggleButton.style.backgroundColor = '#cac5c5';
        })

        toggleButton.addEventListener('mouseout', () => {
            toggleButton.style.backgroundColor = '#ada5a5';
        })

        // Добавляем кнопку в контейнер
        container.appendChild(toggleButton);

        // Возвращаем контейнер с кнопкой
        return container;
    }

    createModalButton(detailsTable) {
        const container = document.createElement('div');
        container.style.display = 'flex';
        container.style.justifyContent = 'flex-end'; // Выравнивание по правому краю
        // Кнопка "Показать подробно/Скрыть"
        const toggleButton = document.createElement('button');
        toggleButton.style.marginLeft = 'auto';
        toggleButton.textContent = 'Детализация в модальном окне';

        toggleButton.addEventListener('click', () => {
            const clone = detailsTable.cloneNode(true);
            clone.style.display = 'block';
            this.modal.showByElement(clone);;
        });

        // Добавляем кнопку в контейнер
        container.appendChild(toggleButton);

        // Возвращаем контейнер с кнопкой
        return container;
    }

}
