    import Dropdown from "../../../ui/dropdown/Dropdown.js";
import Toggle from "../../../ui/toggle/toggle.js";


/**
 * Класс, представляющий элемент управления таблицей планирования канала.
 */
export default class ControlPanel {

    /**
     * Конструктор класса.
     * @param {string} selector - Селектор элемента, в который будет вставлен элемент управления.
     * @param {ChannelPlanTable} table - Таблица, с которой будет связан элемент управления.
     */
    constructor(selector, table) {
        // Получаем элемент, в который будет вставлен элемент управления.
        this.element = document.querySelector(selector);
        // Создаем элемент, который будет являться элементом управления.
        this.panelElement = this.element.appendChild(document.createElement('div'));
        // Добавляем класс, определяющий стиль элемента управления.
        this.panelElement.classList.add('channel-plan-control-panel');

        // Кнопка обновления
        this.refreshButton = this.panelElement.appendChild(document.createElement('button'));
        // Устанавливаем текст и заголовок кнопки.
        this.refreshButton.textContent = '↻ Обновить';
        this.refreshButton.title = "Обновить данные";
        // Добавляем класс, определяющий стиль кнопки.
        this.refreshButton.classList.add('channel-plan-control-panel__button');
        // Создаем элемент для выпадающего списка выбора КА.
        const selectKAElement = this.panelElement.appendChild(document.createElement('div'));

        // Выпадающий список выбора КА.
        this.dropdownKA = new Dropdown(selectKAElement, 'Выберите КА');

        // Связываем выпадающий список с таблицей и выбираем при изменении
        // то есть перестраиваем таблицу.
        this.dropdownKA.onDropdownChange((event) => {
            table.setSelectKa(event.detail);
        });

        // Создаем toggle переключатель режима
        this.toggle = new Toggle('Автообновление');
        // Сохраняем интервал, для того чтобы затем его удалить.
        this.refreshInterval  =  null ;
        this.refreshDelay = 1000;
        // Добавляем что мы 
        this.panelElement.appendChild(this.toggle.element);
        this.toggle.onChange((event) => {
            const isChecked = event.target.checked;
            this.setAutorefresh(isChecked);
        })

        // Устанавливаем таблицу, с которой будет связан элемент управления.
        this.table = table;
    }

    setAutorefresh(isAutorefresh) {
        if (isAutorefresh) {
            this.toggle.getInput().checked = true;
            if (this.refreshInterval !== null ) return
            this.refreshInterval = setInterval(() => {
                this.refresh();
            }, this.refreshDelay);
        } else {
            this.toggle.getInput().checked = false;
            if (this.refreshInterval === null ) return
            clearInterval(this.refreshInterval);
            this.refreshInterval = null;
        }
    }

    /**
     * Добавляем пункты в выпадающий список выбора КА.
     * @param {Array} kaNames - Массив названий КА.
     */
    seKatNames(kaNames) {
        const newDict = {};
        // Создаем словарь, где ключ - название КА, а значение - название КА.
        kaNames.forEach((kaName) => {
            newDict[kaName] = kaName;
        });
        // Устанавливаем словарь в выпадающий список выбора КА.
        this.dropdownKA.setFromDict(newDict);
        // Выбираем все элементы в выпадающем списке.
        this.dropdownKA.selectAll();
    }

    /**
     * Устанавливаем обработчик события нажатия кнопки обновления.
     * @param {Function} callback - Функция, вызываемая при нажатии кнопки обновления.
     */
    onRefresh(callback) {
        this.refreshButton.onclick = () => {
            callback();
        };
    }

    refresh() {
        this.refreshButton.click();
    }
}
