export function showRssPlanAntenDiagram(start, end, data) {
    const dataDiograms = []
    // ID_KA-ID_RSS
    const cashBlocks = {}
    // ID_KA
    const cashLines = {}
    data.forEach(rssSeans => {
        const {
            ID,
            ID_KA,
            Data_Time_In1,
            Shirota_KA1,
            Dolgota_KA1,
            H_KA1,
            Data_Time_In2,
            Shirota_KA2,
            Dolgota_KA2,
            H_KA2,
            Data_Time_Out,
            ID_RSS,
            ID_RSS_Ant,
            Shirota_RSS,
            Dolgota_RSS,
            H_RSS,
            Azimut1,
            UM1,
            Azimut2,
            UM2,
            ID_Err,

            KA_NAME,
            RSS_NAME
        } = rssSeans;


        const id_seans = `${ID_KA}-${ID_RSS}`
        // Создаем группу если нету КА
        if (!(ID_KA in cashLines)) {
            const lines = []
            const group = {
                name: `<span style="font-size: 16px">${KA_NAME}</span>`,
                styles: {},
                classes: [],
                lines: lines
            }
            dataDiograms.push(group)
            cashLines[ID_KA] = lines;
        }
        const lines = cashLines[ID_KA]
        // Если нету блока создаем линию
        if (!(id_seans in cashBlocks)) {
            const blocks = []
            const line = {
                nameElement: {
                    name:  `РСС ${RSS_NAME} <br><span>   Широта: ${Shirota_RSS} <br>Долгота: ${Dolgota_RSS} <br> Антенна: ${ID_RSS_Ant} </span>`,

                    classes: [],

                    caption: '...',
                    styles: {},
                    classes: [],
                },
                emptySetting: {
                    name: 'Нет связи',
                    styles: {},
                    classes: [],
                },
                blocksSetting: {
                    styles: {},
                    classes: [],
                    allStyles: {},
                    allClasses: [],
                },
                blocks: blocks
            }
            lines.push(line)
            cashBlocks[id_seans] = blocks
        }

        
        const blocks = cashBlocks[id_seans]

        // angle
        const block = {
            start: new Date(Data_Time_In1),
            end: new Date(Data_Time_Out),

    })
    rssPlanAntenDiagram.setFormatData(start, end, dataDiograms);
}

        // Проходим по каждому спутнику (КА)
        station.lines.forEach(line => {
            const satelliteName = line.nameElement.name;  // Название спутника (КА)

            // Если такой КА еще нет в результатах, создаем его
            if (!result[satelliteName]) {
                result[satelliteName] = {
                    name: `<span style="font-size: 16px">${satelliteName}</span>`,
                    lines: []
                };
            }

            // Добавляем РСС к этому КА
            result[satelliteName].lines.push({
                nameElement: { name: stationName, styles: line.nameElement.styles },  // Название РСС
                emptySetting: line.emptySetting,
                captionElement: line.captionElement,
                blocks: line.blocks
            });
        });
    });

    // Преобразуем объект в массив
    const resultArray = Object.values(result);

    // Сортируем массив по названиям спутников
    resultArray.sort((a, b) => a.name.localeCompare(b.name));

    // Возвращаем отсортированный массив
    return resultArray;
}
