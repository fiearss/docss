    ka/modules/animations/animations.js

    import { GLOBAL_PERIOD } from "../state/state.js";
import { Animation } from "../../utils/animateUtils.js";
import dateUtils from "../../utils/dateUtils.js";

/**
 * Элементы интерфейса
 */
const speedLabel = document.querySelector('#panel-1-speed'); // Получаем ссылку на label скорости
const speedRange = document.querySelector('#panel-1-speed-range'); // Получаем ссылку на диапазон скорости
// Получаем ссылку на таймлайн и label текущей даты
const timeline = document.querySelector('#panel-1-timeline');
timeline.disabled = true; // Отключаем таймлайн
const currentDateLabel = document.querySelector('#panel-1-date-current'); // Получаем ссылку на label текущей даты
const btnStartAnimation = document.querySelector('#startAnimation'); // Получаем ссылку на кнопку анимации
btnStartAnimation.disabled = true; // Отключаем кнопку анимации

/**
 * Инициализации анимации
 */
// Создаем класс для работы с анимацией спутников
export const animate = new Animation(30, 1000); // Создаем новый объект класса Animation с параметрами шаг анимации и период анимации

/** 
 * Обработчики (связь с анимацией)
*/
// Добавляем обработчик изменения значения скорости
speedRange.addEventListener('input', () => {
    speedLabel.textContent = speedRange.value; // Устанавливаем значение label скорости
    animate.setSpeed(speedRange.value); // Устанавливаем скорость анимации
});

// Добавляем обработчик изменения значения таймлайна
timeline.addEventListener('change', () => {
    const date = new Date(parseInt(timeline.value)); // Создаем объект Date с значением таймлайна
    animate.setCurrentDate(date); // Устанавливаем текущую дату анимации
});

// Добавляем обработчик нажатия на кнопку анимации
btnStartAnimation.addEventListener('click', () => {
    if (animate.isAnimating()) { // Проверяем, есть ли анимация
        animate.stop(); // Останавливаем анимацию
    } else {
        animate.start(); // Запускаем анимацию
    }
});

// Обработчик при создании анимации
animate.onInit(() => {
    btnStartAnimation.disabled = false; // Включаем кнопку анимации
    timeline.disabled = false; // Включаем таймлайн
    // Задаем таймлайн
    timeline.max = GLOBAL_PERIOD.end.getTime(); // Устанавливаем максимальное значение таймлайна
    timeline.min = GLOBAL_PERIOD.start.getTime(); // Устанавливаем минимальное значение таймлайна
});

// Обработчик при удалении анимации
animate.onDelete(() => {
    btnStartAnimation.textContent = 'Возобновить'; // Устанавливаем текст кнопки "Возобновить"
    currentDateLabel.textContent = '00.00.0000 00:00:00'; // Устанавливаем текст label текущей даты
    btnStartAnimation.disabled = true; // Отключаем кнопку анимации
    timeline.disabled = true; // Отключаем таймлайн
});

// Обработчик изменения состояния анимации
animate.onChange((isAnimating) => {
    if (isAnimating) { // Проверяем, есть ли анимация
        btnStartAnimation.textContent = 'Пауза'; // Устанавливаем текст кнопки "Пауза"
    } else {
        btnStartAnimation.textContent = 'Возобновить'; // Устанавливаем текст кнопки "Возобновить"
    }
});

// Работает при каждой итерации анимации
animate.addCallback((currentDate) => {
    // Задаем текущую дату анимации
    timeline.value = currentDate.getTime(); // Устанавливаем значение таймлайна
    currentDateLabel.textContent = dateUtils.formatDisplayDateTime(currentDate); // Устанавливаем текст label текущей даты
})
