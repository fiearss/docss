/**
 * Генерирует чекбоксы для выбора орбит.
 *
 * @param {Tabulator} table - Таблица, из которой берутся данные.
 * @param {string} containerClass - Класс контейнера для чекбоксов.
 */
export function generateOrbitCheckboxes(table = kaTable, containerClass = 'choice-orbit') {
    // Получаем данные из таблицы
    const tableData = table.getData();

    // Извлекаем значения из столбца "orbit" и собираем их в массив
    const orbitValues = tableData.map(row => row.orbit);

    // Фильтруем уникальные значения
    const uniqueOrbits = [...new Set(orbitValues)];

    // Получаем элемент контейнера для чекбоксов
    const choiceOrbitContainer = document.querySelector(`.${containerClass}`);

    // Очищаем контейнер перед добавлением новых элементов
    choiceOrbitContainer.innerHTML = '';

    // Создаем и вставляем чекбоксы
    uniqueOrbits.forEach(orbit => {
        const checkboxLabel = document.createElement('label');
        const checkboxInput = document.createElement('input');

        checkboxInput.type = 'checkbox';
        checkboxInput.checked = true;
        checkboxInput.classList.add(`checkbox-orbit-${orbit}`);

        checkboxLabel.appendChild(checkboxInput);
        checkboxLabel.appendChild(document.createTextNode(`Орбита ${orbit}`));

        choiceOrbitContainer.appendChild(checkboxLabel);

        // Добавляем обработчик изменения состояния чекбокса
        checkboxInput.addEventListener('change', (event) => {
            // Получаем все строки, относящиеся к данной орбите
            const selectedRows = tableData.filter(row => row.orbit === orbit).map(row => row.id);

            if (event.target.checked) {
                table.selectRow(selectedRows);
            } else {
                table.deselectRow(selectedRows);
            }
        });
    });

    /**
     * Обновляет состояния чекбоксов.
     */
    function updateCheckboxes() {
        const selectedData = table.getSelectedData();

        uniqueOrbits.forEach(orbit => {
            const checkbox = document.querySelector(`.checkbox-orbit-${orbit}`);
            const selectedRows = selectedData.filter(row => row.orbit === orbit).map(row => row.id);
            const allRowsForOrbit = tableData.filter(row => row.orbit === orbit).map(row => row.id);

            // Если все строки для данной орбиты выбраны, то чекбокс включен, иначе выключен
            checkbox.checked = allRowsForOrbit.every(id => selectedRows.includes(id));
        });
    }

    // Обновляем состояния чекбоксов при создании
    updateCheckboxes();

    // Добавляем обработчик изменения выбора строк в таблице
    table.on("rowSelectionChanged", function (data, rows) {
        updateCheckboxes();
    });
}
