    /**
 * Модуль `dateUtils` содержит утилиты для работы с датами.
 *
 * @module dateUtils
 * @exports padTo2Digits
 * @exports setInputDate
 * @exports setInputTime
 * @exports getDateFromInputs
 * @exports formatDisplayDate
 * @exports formatDisplayTime
 */

/**
 * Форматирует число, чтобы у него было не менее 2 знака.
 *
 * @param {number} num - Число, которое будет форматировано.
 * @return {string} - Форматированное число в виде строки.
 */
const padTo2Digits = (num) => {
    return num.toString().padStart(2, '0');
}

/**
 * Установить значение элемента ввода типа 'date' из объекта JavaScript Date.
 *
 * @param {HTMLInputElement} dateInput - Элемент ввода типа 'date'.
 * @param {Date} date - Объект JavaScript Date.
 * @return {void}
 */
const setInputDate = (dateInput, date) => {
    dateInput.value = [date.getFullYear(), padTo2Digits(date.getMonth() + 1), padTo2Digits(date.getDate()),].join('-')
}

/**
 * Установить значение элемента ввода типа 'time' из объекта JavaScript Date.
 *
 * @param {HTMLInputElement} timeInput - Элемент ввода типа 'time'.
 * @param {Date} date - Объект JavaScript Date.
 * @return {void}
 */
const setInputTime = (timeInput, date) => {
    // Извлекаем часы, минуты и секунды из объекта даты и форматируем их как строку с ведущими нулями, если это необходимо.
    // Например, 1:2:3 превращается в 01:02:03.
    timeInput.value = [
        padTo2Digits(date.getHours()),
        padTo2Digits(date.getMinutes()),
        padTo2Digits(date.getSeconds()),
    ].join(':');
}


/**
 * Возвращает объект Date, созданный из значений ввода даты и времени.
 *
 * @param {HTMLInputElement} dateInput - Элемент ввода типа 'date'.
 * @param {HTMLInputElement} timeInput - Элемент ввода типа 'time'.
 * @return {Date} Объект Date, созданный из значений ввода даты и времени.
 */
const getDateFromInputs = (dateInput, timeInput) => {
    // Формируем строку в формате 'yyyy-mm-ddThh:mm:ss', и создаем объект Date из нее.
    // Пример: 2020-06-20T15:30:45
    return new Date(dateInput.value + 'T' + timeInput.value);
}


/**
 * Возвращает строку, представляющую отображаемую дату в формате 'dd.mm.yyyy'.
 * 
 * @param {Date} date - Объект JavaScript Date.
 * @return {string} Строка, представляющая отображаемую дату.
 */
const formatDisplayDate = (date) => {
    // Формируем объект с опциями локализации даты.
    // 'day': '2-digit' - дни в формате 2 цифры,
    // 'month': '2-digit' - месяцы в формате 2 цифры,
    // 'year': 'numeric' - год в числовом формате.
    const options = {
        day: '2-digit',
        month: '2-digit',
        year: 'numeric'
    };

    // Преобразуем объект Date в строку с использованием опций локализации.
    // Результат представляет собой отображаемую дату в формате 'dd.mm.yyyy'.
    return date.toLocaleDateString([], options);
}


/**
 * Возвращает строку, представляющую отображаемое время в формате 'hh:mm:ss'.
 *
 * @param {Date} date - Объект JavaScript Date.
 * @return {string} Строка, представляющая отображаемое время.
 */
const formatDisplayTime = (date) => {
    // Формируем объект с опциями локализации времени.
    // 'hour': '2-digit' - часы в формате 2 цифры,
    // 'minute': '2-digit' - минуты в формате 2 цифры,
    // 'second': '2-digit' - секунды в формате 2 цифры.
    const options = {
        hour: '2-digit',
        minute: '2-digit',
        second: '2-digit'
    };

    // Преобразуем объект Date в строку с использованием опций локализации.
    // Результат представляет собой отображаемое время в формате 'hh:mm:ss'.
    return date.toLocaleTimeString([], options);
}

/**
 * Возвращает строку, представляющую отображаемую дату и время в формате 'dd.mm.yyyy, hh:mm:ss'.
 *
 * @param {Date} date - Объект JavaScript Date.
 * @param {string} prefix - Разделитель между датой и временем.
 * @return {string} Строка, представляющая отображаемую дату и время.
 */
const formatDisplayDateTime = (date, prefix=', ') => {
    // Формируем отображаемую дату с помощью функции formatDisplayDate.
    const formattedDate = formatDisplayDate(date);

    // Формируем отображаемое время с помощью функции formatDisplayTime.
    const formattedTime = formatDisplayTime(date);

    // Возвращаем строку, содержащую отображаемую дату и время, разделенные запятой.
    return formattedDate + prefix + formattedTime;
}

/**
 * Возвращает общее количество секунд в течение, указанном в элементе input.
 *
 * @param {HTMLInputElement} timeInput - Элемент input с временем.
 * @return {number} Общее количество секунд в течение.
 */
const getInputTotalSeconds = (timeInput) => {
    const [hours, minutes, seconds] = timeInput.value.split(':').map(Number);
    return (hours * 3600) + (minutes * 60) + seconds;
}

/**
 * Устанавливает значение элемента input типа 'time' из общего количества секунд.
 *
 * @param {HTMLInputElement} timeInput - Элемент input с временем.
 * @param {number} totalSeconds - Общее количество секунд в течение.
 */
const setInputTotalSeconds = (timeInput, totalSeconds) => {
    // Разделение общего количества секунд на часы, минуты и секунды.
    const hours = Math.floor(totalSeconds / 3600); // Получение числа часов.
    const minutes = Math.floor((totalSeconds - (hours * 3600)) / 60); // Получение числа минут.
    const seconds = totalSeconds - (hours * 3600) - (minutes * 60); // Получение числа секунд.

    // Форматирование чисел в формате 2 цифры с ведущими нулями.
    const formattedTime = [hours, minutes, seconds]
        .map(n => n.toString().padStart(2, '0')) // Форматирование числа в формате 2 цифры.
        .join(':'); // Объединение чисел в строку с использованием двоеточия.

    // Установка значения элемента input.
    timeInput.value = formattedTime;
}



/**
 * Форматирует разницу во времени в удобочитаемый текст.
 * 
 * @param {number} timeDiff - Разница во времени в миллисекундах.
 * @return {string} Отформатированный текст, представляющий разницу во времени.
 */
function formatDisplayTextTimeDifference(timeDiff) {
    const seconds = Math.floor(timeDiff / 1000);

    if (seconds < 60) {
        return `менее минуты назад`;
    }
    const minutes = Math.floor(seconds / 60);
    const hours = Math.floor(minutes / 60);
    const days = Math.floor(hours / 24);
    const weeks = Math.floor(days / 7);

    if (weeks > 0) {
        return `${weeks} недель назад`;
    } else if (days > 0) {
        return `${days} дней назад`;
    } else if (hours > 0) {
        return `${hours} часов назад`;
    } else {
        return `${minutes} минут назад`;
    }
}

/**
 * Преобразует секунды в формат часы:минуты:секунды.
 *
 * @param {number} seconds - Количество секунд.
 * @return {string} Форматированный строк в формате часы:минуты:секунды.
 */
function secondsToHMS(seconds) {
    const h = String(Math.floor(seconds / 3600)).padStart(2, '0');
    const m = String(Math.floor(seconds % 3600 / 60)).padStart(2, '0');
    // const s = String(seconds % 60).padStart(2, '0');
    const s = String(Math.floor(seconds % 60)).padStart(2, '0');
    const ms = (seconds % 1).toFixed(3).substring(2); // Получаем миллисекунды с 3 знаками после запятой
    return `${h}:${m}:${s}`;
}

/**
 * Модуль `dateUtils` содержит утилиты для работы с датами и временем.
 * Объединение в модуль (на подобии класса)
 *
 * @module dateUtils
 * @exports padTo2Digits
 * @exports setInputDate
 * @exports setInputTime
 * @exports getDateFromInputs
 * @exports formatDisplayDate
 * @exports formatDisplayTime
 * @exports formatDisplayDateTime
 * @exports getInputTotalSeconds
 * @exports setInputTotalSeconds
 * @exports formatDisplayTextTimeDifference
 * @exports secondsToHMS
 */
const dateUtils = {
    padTo2Digits,
    setInputDate,
    setInputTime,
    getDateFromInputs,
    formatDisplayDate,
    formatDisplayTime,
    formatDisplayDateTime,
    getInputTotalSeconds,
    setInputTotalSeconds,
    formatDisplayTextTimeDifference,
    secondsToHMS
};

export default dateUtils;
