export class Animation {
    /**
     * Представляет объект анимации.
     *
     * @constructor
     * @param {number} speed - Скорость анимации. Если равно 2, то анимация будет проигрываться в два раза быстрее, чем задержка между кадрами.
     * @param {number} delay - Задержка между кадрами анимации. - в миллисекундах (мс)
     */
    constructor(speed, delay) {
        /**
         * Скорость анимации.
         * @type {number}
         */
        this.speed = speed;

        /**
         * Задержка между кадрами анимации. - в миллисекундах (мс)
         * @type {number}
         */
        this.delay = delay;

        /**
         * Начальная дата анимации.
         * @type {Date}
         */
        this.startDate = null;

        /**
         * Конечная дата анимации.
         * @type {Date}
         */
        this.endDate = null;

        /**
         * Текущая дата анимации.
         * @type {Date}
         */
        this.currentDate = null;

        /**
         * Функция обратного вызова, вызываемая на каждом кадре анимации.
         * @type {Function}
         */
        this.callback = null;

        /**
         * Список функций обратного вызова, вызываемых на каждом кадре анимации.
         * @type {Array.<Function>}
         * 
         */
        this.callbackList = [];

        /**
         * Статус, указывающий, запущена ли в данный момент анимация.
         * @type {boolean}
         */
        this.animating = false;  // Указывает, выполняется ли в данный момент анимация

        /**
         * ID таймера анимации.
         * @type {number}
         */
        this.timerId = null;    // ID таймера

        /**
         * Функция обратного вызова, вызываемая при изменении анимации.
         * @type {Function}
         */
        this.onChangeCallback = () => { }; // Функция обратного вызова при изменении анимации

        /**
         * Функция обратного вызова, вызываемая при удалении анимации.
         * @type {Function}
         */
        this.deleteCallback = () => { }; // Функция обратного вызова при удалении анимации

        /**
         * Функция обратного вызова, вызываемая при инициализации анимации.
         * @type {Function}
         */
        this.initCallback = () => { }; // Функция обратного вызова при инициализации анимации
    }


    /**
     * Инициализирует анимацию с заданными датами и функцией обратного вызова.
     *
     * @param {Date} startDate - Начальная дата анимации.
     * @param {Date} endDate - Конечная дата анимации.
     * @param {Function} callback - Функция обратного вызова, вызываемая на каждом кадре анимации.
     */
    init(startDate, endDate, callback) {
        // Начальная дата анимации.
        this.startDate = startDate;

        // Конечная дата анимации.
        this.endDate = endDate;

        // Текущая дата анимации.
        this.currentDate = new Date(startDate);

        // Функция обратного вызова, вызываемая на каждом кадре анимации.
        this.callback = callback;

        // Вызываем функцию инициализации анимации.
        this.initCallback();
    }

    /**
     * Добавляет функцию обратного вызова в список функций.
     * Эта функция будет вызываться при каждом кадре анимации.
     *
     * @param {Function} callback - Функция обратного вызова.
     * Которая принимает: 
     * - `currentDate: {Date}`  - Текущая дата анимации.
     */
    addCallback(callback) {
        // Добавляем новую функцию в список функций обратного вызова.
        // Эта функция будет вызываться при каждом кадре анимации.
        this.callbackList.push(callback);
    }

    /**
     * Вызывает все функции обратного вызова с текущей датой анимации.
     * Это полезно для обновления нескольких компонентов с одной и той же датой анимации.
     */
    triggerCallbacks() {
        /**
         * Проходим по каждой функции обратного вызова и выполняем ее с текущей датой анимации.
         */
        this.callbackList.forEach(callback => {
            /**
             * Выполняем функцию обратного вызова с текущей датой анимации.
             * @param {Date} currentDate - Текущая дата анимации.
             */
            callback(this.currentDate);
        });
    }



    /**
     * Создает и запускает анимацию.
     *
     * @param {Date} startDate - Начальная дата анимации.
     * @param {Date} endDate - Конечная дата анимации.
     * @param {Function} callback - Функция обратного вызова, вызываемая на каждом кадре анимации. 
     * Которая принимает текущую дату анимации:
     * - `currentDate: {Date}` - Текущая дата анимации.
     */
    create(startDate, endDate, callback) {
        // Удаляем предыдущую анимацию, если есть.
        this.delete();

        // Инициализируем новую анимацию.
        this.init(startDate, endDate, callback);

        // Запускаем таймер для анимации.
        this.timerId = setInterval(() => {
            // Если текущая дата достигла конечной даты, анимация завершается.
            if (this.currentDate >= this.endDate) {
                this.stop();
            }

            // Если анимация активна, выполняем функцию обратного вызова с текущей датой.
            if (this.animating) {
                this.callback(this.currentDate);
                this.triggerCallbacks();
                // Открываем новое время с учетом скорости и задержки.
                this.currentDate = new Date(this.currentDate.getTime() + this.speed * this.delay);
            }
        }, this.delay);
    }
    /**
     * Устанавливает текущую дату анимации.
     *
     * @param {Date} date - Новая дата анимации. 
     */
    setCurrentDate(date) {
        /**
         * Текущая дата анимации.
         * @type {Date}
         */
        this.currentDate = date;
    }

    /**
     * Устанавливает скорость анимации.
     *
     * @param {number} speed - Новая скорость анимации.
     */
    setSpeed(speed) {
        /**
         * Скорость анимации.
         * @type {number}
         */
        this.speed = speed;
    }


    /**
     * Стартует анимацию.
     * Это устанавливает статус анимации в активный и вызывает функцию обратного вызова с этим статусом.
     */
    start() {
        // Устанавливаем статус анимации в активный.
        this.animating = true;

        // Вызываем функцию обратного вызова с этим статусом.
        this.onChangeCallback(this.animating);
    }

    /**
     * Останавливает анимацию.
     * Устанавливает статус анимации в неактивный и вызывает функцию обратного вызова 
     * с этим статусом.
     */
    stop() {
        // Устанавливаем статус анимации в неактивный.
        this.animating = false;

        // Вызываем функцию обратного вызова с этим статусом.
        // notify the callback about the change in animation status
        this.onChangeCallback(this.animating);
    }

    /**
     * Удаляет анимацию.
     * Устанавливает статус анимации в неактивный, останавливает таймер, если он запущен,
     * и вызывает функцию удаления.
     */
    delete() {
        // Устанавливаем статус анимации в неактивный.
        this.animating = false;

        // Если таймер запущен, останавливаем его.
        if (this.timerId) {
            // Останавливаем таймер для анимации.
            clearInterval(this.timerId);
        }

        // Вызываем функцию удаления.
        this.deleteCallback();
    }

    /**
     * Проверяет, активна ли в данный момент анимация.
     *
     * @return {boolean} - True, если анимация активна, false в противном случае.
     */
    isAnimating() {
        /**
         * Указывает, активна ли в данный момент анимация.
         * @type {boolean}
         */
        return this.animating;
    }


    /**
     * Проверяет, была ли создана анимация.
     *
     * @return {boolean} - True, если анимация создана, false в противном случае.
     */
    isCreated() {
        /**
         * Указывает, была ли создана анимация.
         * @type {boolean}
         */
        // Проверяет, не равен ли timerId null, что означает, что анимация была создана.
        return this.timerId !== null;
    }


    /**
     * Устанавливает функцию обратного вызова на изменение.
     * Эта функция будет вызываться при изменении состояния анимации, то есть когда происходит возобновление и пауза анимации.
     *
     * @param {Function} callback - Функция обратного вызова.
     * Которая принимает: 
     * - `animating: {boolean}` - Текущий статус анимации.
     */
    onChange(callback) {
        /**
         * Функция обратного вызова на изменение.
         * @type {Function}
         */
        // Эта функция будет вызываться при каждом изменении свойств анимации.
        this.onChangeCallback = callback;
    }


    /**
     * Устанавливает функцию обратного вызова на удаление.
     * Эта функция будет вызываться при удалении анимации.
     *
     * @param {Function} callback - Функция обратного вызова.
     */
    onDelete(callback) {
        /**
         * Функция обратного вызова на удаление.
         * @type {Function}
         */
        // Эта функция будет вызываться при удалении анимации.
        this.deleteCallback = callback;
    }

    /**
     * Устанавливает функцию обратного вызова при инициализации.
     * Эта функция будет вызываться при инициализации анимации.
     *
     * @param {Function} callback - Функция обратного вызова.
     */
    onInit(callback) {
        /**
         * Функция обратного вызова при инициализации.
         * @type {Function}
         */
        // Это функция будет вызываться при инициализации анимации.
        this.initCallback = callback;
    }

}
